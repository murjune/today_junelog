# 방 탈출

문이 5개가 있고, 각 문마다 퀴즈가 있습니다. (랜덤)  
맞히면 통과, 틀리면 실패!  
방을 탈출하시오.

``` c

# include <stdio.h>
# include <time.h>
# include <stdlib.h>
// 문이 5개가 있고, 각 문마다 퀴즈가 있습니다. (랜덤)
// 맞히면 통과, 틀리면 실패! 방탈출을 그만하고 싶으면 -1입력
// 방을 탈출하시오.

int getRandomNum(int level);
int main(void)
{
    int cnt =0 ;
    srand(time(NULL));
    for(int i = 0; i <5 ; i++)
    {
        // x * y = ?
        int x = getRandomNum(i);
        int y = getRandomNum(i);
        printf("###########문제#################\n\n");
        printf("%d x %d = ? (종료: -1)\n\n", x, y);
        printf("##############################\n\n");
        int ans  = x*y;
        int input;
        scanf("%d", &input);
        if (input == -1)
            {
                printf("게임을 종료하겠습니다.\n");
                exit(0);
            }
        if (ans == input)
        {   
            cnt += 1;
            printf("\n\n정답!\n\n");
        }
        else 
        {
            printf("\n\n땡!\n\n");
        }
    }
    if (cnt < 5)
        {
            printf("당신이 맞춘 문제의 개수는 %d입니다. 수고하셨습니다.\n", cnt);
        }
    else 
        {
            printf("모든 문제를 맞추었습니다! 축하드립니다!\n") ;
        }


    return 0 ;  
}
int getRandomNum(int level)
{
    int ret = rand() % 7*(level+1) + 1;
    return ret ; 
}
```
