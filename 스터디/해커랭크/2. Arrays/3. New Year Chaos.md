# 풀이 1- merge sort  O(nlogn)
```
swap의 총 합을 묻는 문제이고, 한 원소가 최대 2번 swap을 할 수 있다. - 2번을 넘어가게 되면 'Too chaotic'를 print해 준다.
```
``` python

import sys
input = lambda : sys.stdin.readline().rstrip()
sys.setrecursionlimit(10**6)

def merge_sort(arr):
    if len(arr) < 2:
        return arr
    mid = len(arr)//2

    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    return merge(left,right)

def merge(left,right):
    global cnt
    from collections import deque
    result = []
    left = deque(left)
    right = deque(right)


    while left or right:
        if left and right:
            if left[0] > right[0]: # swap이 발생
                v= right.popleft()
                cnt += len(left) # swap한 횟수 더해주기
                result.append(v)
            elif left[0] <= right[0]:
                v = left.popleft()
                result.append(v)

        elif left and not right:
            result.append(left.popleft())
        elif not left and right:
            result.append(right.popleft())

    return result

t = int(input())
for _ in range(t):
    n = int(input()) # 10만 - 최대 O(nlogn) 알고리즘 풀이

    start = [x for x in range(1, n+1)]
    end = list(map(int, input().split()))
    cnt = 0

    for i in range(n):
        if (end[i]-1) - i > 2:
            print('Too chaotic')
            break
    else:
        merge_sort(end)
        print(cnt)




```

# 풀이 2 - O(N)
```
일반적인 inversion Counting문제였다면, O(N^2)의 시간복잡도를 갖게 되겠지만  
한 원소가 최대 2번까지만 뇌물을 줄 수 있기 때문에 O(N*2) = O(N)의 시간 복잡도가 나온다.
```

def minimumBribes(n, q):
    for i, j in enumerate(q):
        if j - i > 2: 
            print("Too chaotic")
            return 
    cnt = 0
    for i in range(1, n): # O(N)
        j = i
        while j > 0 and q[j-1] > q[j]:  # O(2): inversion count는 최대 2번까지 이다. 그 이상은 위 코드에서 걸러진다.
            q[j-1], q[j] = q[j], q[j-1]
            j -= 1
            cnt+=1
    print(cnt)
