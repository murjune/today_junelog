def jumpingOnClouds(c):

    # 최단 경로 이므로 bfs로 풀면 간단히 해결 가능 
    from collections import deque
    visited = [-1] * n
    visited[0] = 0
    q = deque([0])

    while q:
        v = q.pop()
        if v == n - 1:
            return (visited[v])
        for Next in [(v + 1), (v + 2)]:
            if Next <= n - 1:  # 범위 안에 있을 때

                if not c[Next] and visited[Next] == -1:  # not 번개 구름 and 방문한 적 없을 때
                    visited[Next] = visited[v] + 1
                    q.append(Next)
n = int(input())
