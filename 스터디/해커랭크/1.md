# 정답

left / cur / right  3분할 하여 생각했다.
``` python
import sys
input = lambda : sys.stdin.readline().rstrip()
from collections import defaultdict

n,r = map(int,input().split())
arr = list(map(int,input().split()))


D = defaultdict(int)
D2 = defaultdict(int)
for num in arr:
    D[num] += 1

ans = 0
for num in arr:
    D[num] -= 1
    if num % r == 0:
        ans += D2[num//r]*D[num*r]
        
    D2[num] += 1

print(ans)

```
# 오답
```python
import sys
input = lambda : sys.stdin.readline().rstrip()
from collections import defaultdict

n,r = map(int,input().split()) 
arr = list(map(int,input().split()))


D = defaultdict(int)
for a in arr:
    D[a] += 1

new_arr = list(D.keys())
new_arr.sort()

ans = 0
for num in new_arr:
    i ,j ,k = num//r, num, num*r
    if i < j < k:
        ans += D[num//r] * D[num] * D[num*r]
    elif i == j == k:
        ans += D[num] *(D[num]-1)*(D[num]-2) // 6
print(ans)
```
