```python
import sys
input = lambda : sys.stdin.readline().rstrip()
from collections import defaultdict

n,r = map(int,input().split()) 
arr = list(map(int,input().split()))


D = defaultdict(int)
for a in arr:
    D[a] += 1

new_arr = list(D.keys())
new_arr.sort()

ans = 0
for num in new_arr:
    i ,j ,k = num//r, num, num*r
    if i < j < k:
        ans += D[num//r] * D[num] * D[num*r]
    elif i == j == k:
        ans += D[num] *(D[num]-1)*(D[num]-2) // 6
print(ans)
```
