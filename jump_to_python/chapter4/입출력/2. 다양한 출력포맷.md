
# 출력 포맷


``` python
# 다양한 출력 포맷 

print("{0}".format(500)) # 500
print("{0}".format(+500))# 500 , +가 출력 안됨
print("{0}".format(-500)) # -500

print("{0:+}".format(500)) # +500
print("{0:+}".format(+500)) # +500
print("{0:+}".format(-500)) # -500

print("{0: >10}".format(500))#        500

# 빈 자리는 비워두기, 우측 정렬, +기호, 10칸공간 확보
print("{0: >+10}".format(500))  #       +500

# 빈 자리는 _ 로, 좌측 정렬, +기호, 10칸공간 확보
print("{0:_<+10}".format(500))  # +500______

# , : 1,000단위로 콤마를 찍는다.
print("{0: ,}".format(10000000000)) #  10,000,000,000


# 최종  예제 - 빈공간 ^채우기, 왼쪽정렬, +부호, 30공간 확보
print("{0:^<+30,}".format(100000000000)) # +100,000,000,000^^^^^^^^^^^^^^

# 실수값 출력

print("{0: f}".format(5/3)) #  1.666667 
print("{0: 2f}".format(5/3)) # 1.67 # 소수 2번째 자리까지 출력

```
