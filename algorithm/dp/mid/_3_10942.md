# 문제 : [펠린드롬?](https://www.acmicpc.net/problem/10942)  

# 풀이 1
```python
import sys
input = lambda : sys.stdin.readline().rstrip()

n = int(input())
nums = list(map(int,input().split()))

d = [[0]*n for _ in range(n)]

# 1. 홀수 펠롬드롬
# mid 을 기준으로 Loop을 돈다
for mid in range(n):
    d[mid][mid] = 1
    for i in range(1,mid+1):
        l,r = mid-i , mid + i

        # 범위 체크
        if l < 0 or r > n-1: break
        # 그 전 범위 펠롬드롬 & 양 끝의 값이 같으면 -> 펠롬드롬
        if d[l+1][r-1] and nums[l] == nums[r]:
            d[l][r] = 1
        else: break
        
# 짝수 펠롬드롬
for i in range(n-1):
    j = i + 1
    if nums[i] == nums[j]:
        d[i][j] = 1
        for k in range(1,i+1):
            l,r = i-k,j+k
            # 범위 체크
            if l < 0 or r > n - 1: break
            # 그 전 범위 펠롬드롬 & 양 끝의 값이 같으면 -> 펠롬드롬
            if d[l + 1][r - 1] and nums[l] == nums[r]:
                d[l][r] = 1
            else:break
# 출력
m = int(input())
for _ in range(m):
    a,b = map(int,input().split())
    print(d[a-1][b-1])


```
# 풀이 2 - 탑 바텀
```python
import sys
sys.setrecursionlimit(10**5)
input = lambda :sys.stdin.readline().rstrip()
n = int(input())
a = [0]+list(map(int,input().split()))
d = [[-1]*(n+1) for _ in range(n+1)] # 재귀 호출 할 시 테이블 초깃값 -1로 하는게 편함

def dp(s,e):
    if s == e: return 1
    if s+1 == e:
        if a[s]== a[e]: return 1
        else: return 0
    if d[s][e] >= 0: return d[s][e]
    if a[s] != a[e]:
        d[s][e] = 0
        return d[s][e]
    else:
        d[s][e] = dp(s+1,e-1)
        return d[s][e]



m = int(input())
for _ in range(m):
    x,y = map(int,input().split())
    print(dp(x,y))
```
# 풀이 3 - 바텀 -업(길이)
```python
import sys
input = lambda :sys.stdin.readline().rstrip()
n = int(input())
a = [0]+list(map(int,input().split()))
d = [[0]*(n+1) for _ in range(n+1)]
# 길이 1
for i in range(1,n+1):
    d[i][i] = 1
# 길이 2
for i in range(1,n):
    if a[i] == a[i+1]:
        d[i][i+1] = 1

# 길이 3
for i in range(3,n+1):
    # i = e-s + 1
    for s in range(1,n+2-i):
        e = s + i - 1
        if a[s] == a[e]:
            d[s][e] = d[s+1][e-1]
        else:
            d[s][e] = 0


m = int(input())
for _ in range(m):
    x,y = map(int,input().split())
    print(d[x][y])
```
