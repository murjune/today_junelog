# 문제 : [팰린드롬 분할](https://www.acmicpc.net/problem/1509)  

팰린드롬? + 응용문제
# 풀이
```python
import sys
input = lambda :sys.stdin.readline().rstrip()

a = input()
n = len(a)
a = ' '+a
##########################
# 1. 구간 (s,e)에서 팰린드롬이 가능한가 여부 , d테이블에 기록
d = [[0]*(n+1) for _ in range(n+1)]
# 길이 1
for i in range(1,n+1):
    d[i][i] = 1
# 길이 2
for i in range(1,n):
    if a[i] == a[i+1]:
        d[i][i+1] = 1

# 길이 3
for i in range(3,n+1):
    # i = e-s + 1
    for s in range(1,n+2-i):
        e = s + i - 1
        if a[s] == a[e]:
            d[s][e] = d[s+1][e-1]
        else:
            d[s][e] = 0

#############################################
d2 = [i for i in range(n+1)]

# 점화식 : d[i] = d[j-1] + 1(j~i까지의 팰린드롬)

for i in range(2,n+1):
    if i == 2:
        if d[1][2]:
            d2[i] = 1
            continue
        else:
            d2[i] = 2
            continue

    for j in range(i):
        l = i-j
        if d[l][i]:
            d2[i] = min(d2[i],d2[l-1]+1)# (l,i) 길이 j짜리 팰린드롬
        else:
            d2[i] = min(d2[i],d2[l]+j)#(l,i) 길이 j not 팰린드롬

print(d2[n])
```
