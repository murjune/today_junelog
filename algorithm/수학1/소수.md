# 소수
# 소수 찾기_1978
https://www.acmicpc.net/problem/1978
``` python


# 풀이
# i (2~ n**0.5)까지 반복하여 나머지가 0인지 알아 본다.
# 1을 예외처리한다.

n = int(input()) # 4

prime_num = list(map(int, input().split())) # 1 3 5 7
answer = 0

for i in prime_num:
    tmp = 0

    if i ==1: # 1일 때를 예외처리 한다.
        continue

    for j in range(2, int(i**0.5)+1):
        if i % j == 0:
            tmp += 1

    if tmp == 0:
        answer += 1

print(answer)






```
