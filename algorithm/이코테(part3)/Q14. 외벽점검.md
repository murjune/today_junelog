def sol1(n, weak, dist):
    from itertools import combinations, permutations
    arr = [0] * n
    for w in weak:
        arr[w] = 1  # 취약한 부분 check

    d = len(dist)
    for i in range(1, d + 1):
        # 투입 시킬 친구 뽑기
        for friends in list(permutations(dist, i)):
            for weak_points in list(combinations(weak, i)):
                # 친구들 weak 부분에 한명 씩 투입
                graph = arr[:]
                for j in range(i):
                    now = weak_points[j] # 탐색 시작 위치
                    move = friends[j]
                    tmp = 0
                    while move >= tmp:
                        if graph[(now + tmp)%n] == 1:
                            graph[(now + tmp) % n] = 0
                        tmp +=1
                if 1 in graph:
                    continue
                else:
                    return i
    return -1
                
                    

                        
                

def sol2(n, weak, dist):
    from itertools import combinations, permutations
    arr = [0] * n
    for w in weak:
        arr[w] = 1  # 취약한 부분 check

    d = len(dist)
    for i in range(1, d + 1):
        # 투입 시킬 친구 뽑기
        for friends in list(permutations(dist, i)):
            for weak_points in list(combinations(weak, i)):
                # 친구들 weak 부분에 한명 씩 투입
                graph = arr[:]
                
                now = weak_points[j] # 탐색 시작 위치
                for j in range(i):
                    move = friends[j]
                    tmp = 0
                    while move >= tmp:
                        if graph[(now - tmp)%n] == 1:
                            graph[(now - tmp) % n] = 0
                        tmp +=1
                if 1 in graph:
                    continue
                else:
                    return i

    return -1

def solution(n, weak, dist):
    result1 = sol1(n, weak, dist)
    result2 = sol2(n, weak, dist)
    if result1 == -1 and result2 == -1:
        return -1
    elif result2 == -1:
        return result1
    elif result1 == -1:
        return result2
    else:
        return min(result1,result2)
