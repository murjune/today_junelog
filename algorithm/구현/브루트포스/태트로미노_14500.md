# 풀이
# 모든 경우의 수를 확인하는 완전 탐색 문제이다.
# 각 도형의 함수를 모두 구하자

# 1. 2 X2 도형

def square(a):
    ans = 0
    for i in range(n):
        for j in range(m):
            if i <= n-2 and j <= m-2:
                sum = (a[i][j]+ a[i][j+1]+a[i+1][j]+a[i+1][j+1])
                ans = max(ans,sum)
    return ans

# 2. 1 X 4 , 4 X 1 도형

def long_square(a):
    ans = 0
    for i in range(n):
        for j in range(m):
            if j <= m-4:
                sum = (a[i][j]+ a[i][j+1]+a[i][j+2]+a[i][j+3])
                ans = max(ans, sum)
            if i <= n-4:
                sum = (a[i][j] + a[i+1][j] + a[i+2][j] + a[i+3][j])
                ans = max(ans, sum)
    return ans
