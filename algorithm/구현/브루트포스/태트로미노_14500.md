# 모든 경우의 수를 확인하는 완전 탐색 문제이다.
# 각 도형의 함수를 모두 구하자

# 1. 2 X2 도형

def square(a):
    ans = 0
    for i in range(n):
        for j in range(m):
            if i <= n-2 and j <= m-2:
                sum = (a[i][j]+ a[i][j+1]+a[i+1][j]+a[i+1][j+1])
                ans = max(ans,sum)
    return ans

# 2. 1 X 4 , 4 X 1 도형

def long_square(a):
    ans = 0
    for i in range(n):
        for j in range(m):
            if j <= m-4:
                sum = (a[i][j]+ a[i][j+1]+a[i][j+2]+a[i][j+3])
                ans = max(ans, sum)
            if i <= n-4:
                sum = (a[i][j] + a[i+1][j] + a[i+2][j] + a[i+3][j])
                ans = max(ans, sum)
    return ans

import sys
input = lambda : sys.stdin.readline().rstrip()


# 입력값
a,b = map(int,input().split())

graph = [list(map(int,input().split())) for _ in range(a)]
visted = [[False for _ in range(b)] for _ in range(a)] # 방문기록
dx = [-1,1,0,0]
dy = [0,0,-1,1]

# dfs 함수 1
def dfs(x,y,sum,cnt):
    global ans

    cnt -= 1
    if cnt == 0:
        if sum > ans:
            ans = sum

        return
    for i in range(4):

        cx = x + dx[i]
        cy = y + dy[i]

        if cx <0 or cy <0 or cx > a-1 or cy > b-1: # 범위 밖 제외
            continue

        if visted[cx][cy] == False:

            # 1. 방문 처리
            visted[cx][cy] = True

            # 3. dfs 시행
            dfs(cx,cy,sum+ graph[cx][cy],cnt) # dfs 시행

            # 4. 방문 다시 되돌리기
            visted[cx][cy] = False

# ㅗ 모양 테트로미노 함수

def fu(x,y,sum):
    global ans
    cnt = 4
    for i in range(4):
        cx = x + dx[i]
        cy = y + dy[i]

        if cx <0 or cy <0 or cx > a-1 or cy > b-1: # 범위 밖 제외
            continue

        cnt -= 1
        sum += graph[cx][cy]

    if cnt == 1: # ㅗ 모양
        ans = max(ans,sum)

    if cnt == 0: # + 모양에서 블록 값을 빼야 할 때

        for i in range(4):
            tmp = sum
            cx = x + dx[i]
            cy = y + dy[i]

            tmp -= graph[cx][cy]
            ans = max(ans, tmp)


# 풀이
ans = 0
for i in range(a):
    for j in range(b):
        visted[i][j] = True # 방문 기록
        dfs(i,j,graph[i][j],4)
        fu(i,j,graph[i][j])
        visted[i][j] = False  # 방문 기록 되돌리기
print(ans)
