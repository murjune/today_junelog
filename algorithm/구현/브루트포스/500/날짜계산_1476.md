# 문제:날짜 계산_1476 -(2)
https://www.acmicpc.net/problem/1476

``` python


# 연도
e,s,m = (map(int, input().split()))# (1 ≤ E ≤ 15, 1 ≤ S ≤ 28, 1 ≤ M ≤ 19)
# 입력

def June_Date(e,s,m):


    (x, y, z) = (0,0,0) # 시작 준규나라의 연도
    year = 0 # 시작 우리의 연도
    while (x, y, z) != (e, s, m):

        (x, y, z) = (x+1, y+1, z+1) # 1년 후 준규나라의 도

        year += 1  # 1년 후 우리의 연도

        if x == 16: # 
            x -= 15

        if y == 29:
            y -= 28

        if z == 20:
            z -= 19

    return year

# 출력
# 첫째 줄에 E S M으로 표시되는 가장 빠른 연도를 출력한다.
print(June_Date(e,s,m))





```
