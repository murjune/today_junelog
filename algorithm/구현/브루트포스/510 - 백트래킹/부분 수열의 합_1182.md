# 문제: 부분 수열의 합
https://www.acmicpc.net/problem/1182

# 풀이
``` python

#  (1 ≤ N ≤ 20, |S| ≤ 1,000,000)
n, s = map(int,input().split())

nums = list(map(int,input().split()))
result = 0
def dfs(nums,tmp):
    global result
    if tmp == s:
        result +=1
    for i in range(len(nums)):
        if i == len(nums)-1:
            dfs([],tmp+nums[i])
        else:
            dfs(nums[i+1:],tmp+nums[i])

dfs(nums,0)
if s == 0:
    result -=1
print(result)



```
