# 문제: 알파벳
https://www.acmicpc.net/problem/1987

그냥 백트래킹(dfs) 문제
# 풀이 
``` python

def dfs(start,cnt):
    global ans
    x, y = start

    for (i,j) in [(x+1,y),(x,y+1),(x-1,y),(x,y-1)]:
        if 0 <=i <= R-1 and 0 <= j <= C-1:
            tmp = graph[i][j]
            if Alpha[tmp] == 0:
                Alpha[tmp] = 1
                dfs((i,j),cnt+1)
                Alpha[tmp] = 0

    ans = max(cnt,ans)
# 풀이
import sys
input = lambda : sys.stdin.readline().rstrip()
from  collections import defaultdict
R, C = map(int,input().split())

graph = [list(input()) for _ in range(R)]

Alpha = defaultdict(int)
Alpha[graph[0][0]] = 1

for i in graph:
    for j in i:
        if j not in Alpha:
            Alpha[j] = 0
ans = 1

dfs((0,0),1)
print(ans)
```
