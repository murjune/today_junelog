# 문제: 부등호
https://www.acmicpc.net/problem/2529


# 풀이 - 재귀호
``` python

import sys
input = lambda : sys.stdin.readline().rstrip()

k = int(input()) # 2 ≤ k ≤ 9
Operators = list(input().split())
pivot = '0123456789'
Min_ans = str(sys.maxsize)
Max_ans = str(-sys.maxsize)

def dfs(cnt,nums):
    global Max_ans,Min_ans

    if cnt == k+1:

        for i in range(k):
            start = nums[i]
            next = nums[i + 1]
            Operator = Operators[i]
            if start < next and Operator != '<':
                return
            elif start > next and Operator != '>':
                return

        tmp = ''.join(map(str, nums))
        # 최댓값
        if int(Max_ans) < int(tmp):
            Max_ans = tmp
        # 최솟값
        if int(Min_ans) > int(tmp):
            Min_ans = tmp
        return

    for i in pivot:
        if i not in nums:
            dfs(cnt+1, nums+i)

dfs(0,'')
print(Max_ans)
print(Min_ans)
```
# 풀이2 - itertools+ 완전탐색

pypy로 풀면 틀린다..
``` python

from itertools import permutations
import sys
input = lambda : sys.stdin.readline().rstrip()

k = int(input()) # 2 ≤ k ≤ 9
Operators = list(input().split())

Min_ans = str(sys.maxsize)
Max_ans = str(-sys.maxsize)

for nums in list(permutations(range(0,10),k+1)):

    flag = True
    for i in range(k):
        start = nums[i]
        next = nums[i+1]
        Operator = Operators[i]
        if start < next and Operator != '<':
            flag = False
            break
        elif start > next and Operator != '>':
            flag = False
            break
    if flag == True:
        tmp = ''.join(map(str,nums))
        # 최댓값
        if int(Max_ans) < int(tmp):
            Max_ans = tmp
        # 최솟값
        if int(Min_ans) > int(tmp):
            Min_ans = tmp

print(Max_ans)
print(Min_ans)
```
