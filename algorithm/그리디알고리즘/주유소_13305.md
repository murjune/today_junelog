# 풀이 1 -17점

dp 풀이
``` python
# 17점
n = int(input()) # 4
d = [0] * (n-1)
road = list(map(int,input().split())) # n-1개 [0,1,2,..,n-2]
cost = list(map(int,input().split())) # n-1개
cost.pop()
cost.reverse()
road.reverse()

road_sum = road[::]
for i in range(1,n-1):
    road_sum[i] += road_sum[i-1]

# print(d)
# print(road)
# print(cost)
# print('도로합:',road_sum)

# 풀이 1 - dp

d[0] = cost[0] * road[0]
for i in range(1,n-1):
    tmp = road_sum[i] * cost[i]
    tmp2 = d[i-1] + road[i]* cost[i]
    d[i] = min(tmp,tmp2)

print(d[n-2])




```
