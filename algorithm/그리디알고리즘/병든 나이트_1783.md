# 문제: 병든 나이트


# 풀이
진짜 가다가다 노가다로 케이스 나눠서 풀었다..
``` python
import sys
input = lambda : sys.stdin.readline().rstrip()

y , x= map(int, input().split())

def count(x,y):
    if y >= 3:
        if x >= 7:
            cnt = 5
            cnt += (x - 7)
            print(cnt)
            return
        if  5<= x <= 6:
            cnt = 4
            print(cnt)
            return
        elif x <=4:
            cnt = x
            print(cnt)
            return
    elif y < 3:
        if y == 2:
            if x >= 7:
                cnt = 4
                print(cnt)
                return
            elif 5 <= x <= 6:
                cnt = 3
                print(cnt)
                return
            elif 3<= x <= 4:
                cnt = 2
                print(cnt)
                return
            elif 1<= x <=2:
                cnt = 1
                print(cnt)
                return
        elif y == 1:
            cnt = 1
            print(cnt)
            return

count(x,y)
```
