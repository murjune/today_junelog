# 오답 - 무엇이 틀렸을까...??

다른 사람들 풀이와 비교해봐도 틀린 부분이 없는 것같은디..??
``` python


# n,m < 3 일 경우 -1 출력

n,m = map(int,input().split())
arr1 = [list(input()) for _ in range(n)]
arr2 = [list(input()) for _ in range(n)]

def switch(x,y): # 연산 함수
    for i in range(x,x+3):
        for j in range(y,y+3):
            if arr1[i][j] == '0':
                arr1[i][j] = '1'
            else:
                arr1[i][j] = '0'

def search():
    cnt = 0
    for i in range(n-2):
        for j in range(m-2):
            if arr1[i][j] == arr2[i][j]:
                continue
            else:
                switch(i,j)
                cnt += 1

    if arr1 == arr2:
        print(cnt)
        return
    else:
        print(-1)
        return

if n and m < 3:
    print(-1)
else:
    search()



```
