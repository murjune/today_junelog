import sys
input = lambda : sys.stdin.readline().rstrip()
print = lambda x: sys.stdout.write(str(x )+ "\n")

t = int(input())
for i in range(1 , t +1):
    I = list(input())
    P = list(input())
    n = len(I)
    m = len(P)

    s = 0
    ans = ""
    cnt = 0
    for c in P:
        if s < n and c == I[s] :
            s+=1
            ans += c
        else:
            cnt +=1

    if len(ans) == n:
        print("Case #{0}: {1}".format(i, cnt))
    else:
        print("Case #{0}: IMPOSSIBLE".format(i))
