def topology_sort():

    from collections import deque

    q = deque()

    for i in range(1,n+1):
        if indegree[i] == 0 :
            q.append(i)

    while q:
        v = q.popleft()

        for i in graph[v]:  # v -> i

            indegree[i] -= 1

            if indegree[i] == 0:
                running_time[i] += running_time[v]
                q.append(i)

# 강의 수 1~ 500, 강의 번호는 1~ n

n = int(input())

graph = [[] for _ in range(n+1)]
indegree = [0] * (n+1)
running_time = [0] * (n+1)

# 그래프, 진입차수 기록
for i in range(1,n+1):
    information = list(map(int,input().split()))
    running_time[i] = information[0]
    L = len(information)

    for j in range(1,L-1):
        graph[information[j]].append(i)
        indegree[i] += 1

# 위상 정렬
topology_sort()

for i, k in enumerate(running_time):
    if i != 0:
        print(k)
