# 위상 정렬
def topology_sort():

    from collections import deque

    q = deque()

    # 진입차수 0 인 과목 q에 넣기
    for i in range(1,n+1):
        if indegree[i] == 0 :
            q.append(i)

    while q:
        now = q.popleft()

        for next in graph[now]:  # v -> i

            indegree[next] -= 1
            ans[next] = max(ans[next], running_time[next]+ ans[now])
      
            if indegree[next] == 0:

                q.append(next)

# 강의 수 1~ 500, 강의 번호는 1~ n
import copy
n = int(input())

# a -> b : 방향 graph
graph = [[] for _ in range(n+1)]

# 진입 차수
indegree = [0] * (n+1)
running_time = [0] * (n+1)

# 그래프, 진입차수 기록
for i in range(1,n+1):
    information = list(map(int,input().split()))
    running_time[i] = information[0] # i 강의: 수업 시간
    L = len(information)

    for j in range(1,L-1):
        graph[information[j]].append(i) # 강의 순서
        indegree[i] += 1 # 진입차수


ans = copy.deepcopy(running_time)
# 위상 정렬
topology_sort()

# 출력
for i, k in enumerate(ans):
    if i != 0:
        print(k)
