# 문제: 여행가자

유니온 파인드 or dfs or bfs 연습문제
# 풀이
``` python

def find(x):
    if parent[x] < 0:
        return x
    y = find(parent[x])
    parent[x] = y
    return y

def union(x,y):
    x =  find(x)
    y  = find(y)
    if x== y :
        return
    if parent[x] <= parent[y]:
        parent[x] += parent[y]
        parent[y] = x
    else:
        parent[y] += parent[x]
        parent[x] = y
# 풀이
```
