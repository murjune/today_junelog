# Weighted Union Find 방식
``` python
n= int(input())
parent = [-1 for x in range(n+1)]

def find(x):
    if parent[x] < 0:
        return x
    y = find(parent[x])
    # 경로 압축
    parent[x] = y
    return y

def union(x,y):
    x = find(x) # x의 부모 노드
    y = find(y) # y의 부모 노드
    if x == y:
        return
    # x의 크기가 y보다  때 (ex, -2, -1)
    if parent[x] < parent[y] : #  -2 -1
        parent[x] += parent[y] # parent[x] = -3
        parent[y] = x # parent[y] = x
    else:
        parent[y] += parent[x]
        parent[x] = y
```
