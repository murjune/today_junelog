# 문제: 팀 결성 - 이코테 책298p

유니온 파인드 기본 문제
# 풀이

```
입력 예시
7 8
0 1 3
1 1 7
0 7 6
1 7 1
0 3 7
0 4 2
0 1 1
1 1 1
출력 예시
NO
NO
YES
```
``` python

def find(x):

    if parent[x] < 0 :
        return x

    y = find(parent[x])
    return  y
def union(x,y):
    x = find(x)
    y = find(y)
    if x == y :
        return
    if parent[x] < parent[y]:
        parent[x] += parent[y]
        parent[y] = x
    else:
        parent[y] += parent[x]
        parent[x] = y
# n,m (1~10만)
# n: 학생 수 m : 연산 수
n,m = map(int,input().split())

parent = [-1 for x in range(n+1)]

for _ in range(m):
    a,b,c = map(int,input().split())

    if a == 0:
        union(b,c)
    if a == 1:
        if find(b) == find(c):
            print('YES')
        else:
            print('NO')

```
