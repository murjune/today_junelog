# 문제: 히스토그램에서 가장 큰 직사각형 - (1)
https://www.acmicpc.net/problem/6549  

강의를 참고하여 풀었다..  
혼자 떠올리기 힘든 풀이인 것 같다. 나중에 여러번 풀어보자!
# 풀이 - O(N)의 시간 복잡도
``` python

import sys
input = lambda : sys.stdin.readline().rstrip()
while True:
    arr = list(map(int ,input().split()))
    if arr == [0]:
        break
    n = arr[0]
    arr =arr[1:]
    check = [0] * n
    stack = []

    # 1단계
    for i in range(0, n):

        while stack and stack[-1][1] > arr[i]:  # 작아

            end, h = stack.pop() # 0 2
            if stack:
                check[end] = h * (i-1 - stack[-1][0])
            else:
                check[end] = h * (i-1 + 1)

        stack.append((i, arr[i]))

    # 2단계 - stack 비우기
    # [(1, 1), (4, 1), (5, 3), (6, 3)]
    end = n-1
    while stack:

        index, h = stack.pop()
        if stack:
            check[index] = h * (end - stack[-1][0])
        else:
            check[index] = h * (end + 1)
    print(max(check))


```
# 오답 - 시간 초과.. O(N^2)
``` python

import sys
input = lambda : sys.stdin.readline().rstrip()
while True:
    arr = list(map(int ,input().split()))
    if arr == [0]:
        break
    n = arr[0]
    arr =arr[1:]
    check = [0] * n
    stack = [(0, arr[0])]

    # 1단계
    for i in range(1, n):

        while stack and stack[-1][1] > arr[i]:  # 작아

            end, h = stack.pop()
            idx = end
            while True:

                if arr[idx] >= arr[end]:
                    if idx == 0:
                        check[end] = h * (i - idx)
                        break

                else:
                    check[end] = h * (i - 1 - idx)
                    break
                idx -= 1

        stack.append((i, arr[i]))

    # 2단계 - stack 비우기
    # [(1, 1), (4, 1), (5, 3), (6, 3)]
    end = n-1
    while stack:

        index, h = stack.pop()
        i = index
        while True:

            if arr[i] < h:
                check[index] = h * (end-i)
                break
            elif arr[i] >= h:
                if i == 0:
                    check[index] = h * (end +1)
                    break

                i -= 1

    print(max(check))


```
