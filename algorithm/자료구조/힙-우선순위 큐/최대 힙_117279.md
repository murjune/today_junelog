# 문제: 최대 힙
https://www.acmicpc.net/problem/11279  

최대 힙 구현 문제
# 풀이
``` python

import sys
input = lambda : sys.stdin.readline().rstrip()
n = int(input()) # (1 ≤ N ≤ 100,000)

import heapq

q = []

for _ in range(n):
    x = int(input())

    if x == 0:

        if q:
            print(-heapq.heappop(q))
        else:
            print(0)
    else:
        heapq.heappush(q,-x)


```
