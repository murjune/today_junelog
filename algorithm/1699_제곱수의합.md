# 1699_제곱수의합

## 틀린 답:
예외 경우:
12 = 4 + 4 +4(3),  내가 만든 코드는 12 = 9+1+1+1(4)
``` python
import math

n = int(input())

# n_sq = [0]*100
n_sq = []

def add(a):
    for i in range(a+1):
        if int(math.sqrt(a)) <= i < int(math.sqrt(a)) + 1:
            n_sq.append(i)



for i in range(0, n):
    add(n)
    n -= n_sq[i] ** 2
    if n == 0:
        break
print(n_sq)
print(len(n_sq))
```
