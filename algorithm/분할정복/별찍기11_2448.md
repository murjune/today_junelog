# 문제: 별찍기 11 -(2)
https://www.acmicpc.net/problem/2448

분할 정복 기본 문제

# 풀이
``` python


n = int(input()) # 3 x 2^k
graph = [[' ' for _ in range(2*n)] for _ in range(n)]

# 3*3 모양 함수
def box(x,y):
    y -= 1
    graph[x][y] = '*'
    graph[x+1][y-1:y+2] = ['*',' ','*']
    graph[x+2][y-2:y + 3] = ['*','*','*','*','*']

def star(x, y, n):

    if n == 3:
        box(x, y)
        return
    n //= 2
    star(x,y,n)
    star(x + n,y - n,n)
    star(x + n,y + n,n)

star(0, n, n)
for i in graph:
    print(''.join(i))

```
