# 문제: 텀 프로젝트
https://www.acmicpc.net/problem/9466


# 오답 - 시간초과

``` python


# 테스트 케이스
import sys
sys.setrecursionlimit(10**6)
input = lambda : sys.stdin.readline().rstrip()


T = int(input())
for _ in range(T):
    # 입력값: 학생수 n (2 ~ 10만)

    n = int(input())
    graph = [0] + list(map(int, input().split()))



    # dfs 함수
    cnt = 0

    def dfs(x):
        global cnt
        visted = [0] * (n + 1)
        v = x
        while True:

            x = graph[x]

            visted[x] += 1 # 방문

            if visted[x] == 2:
                if visted[v] == 0:
                    cnt +=1
                break

    for i in range(1,n+1):
        dfs(i)

    print(cnt)

```
