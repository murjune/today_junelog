# 문제: 중량제한_1939
https://www.acmicpc.net/problem/1939


``` python
import sys
sys.setrecursionlimit(10000)

n, m = map(int ,input().split())


graph = [[]for _ in range(n+1)]

for _ in range(m):
    a, b, c = map(int,input().split()) # c: 중량

    graph[a].append([b,c])
    graph[b].append([a,c])

x, y = map(int,input().split()) # 시작 섬, 끝 섬

visted = [False] * (n+1) # 방문기록


# [[0], [[2, 2], [3, 3]], [[1, 2], [3, 2]], [[1, 3], [2, 2]]]

def dfs(x,y,k): # x: 시작 섬, y: 끝 섬, k: 중량 기준
    if x == y:
        return True

    visted[x] = True # 방문 처리

    for i in graph[x]:
        if visted[i[0]] == True: # 이미 방문한 곳이면 안됨
            continue
        if i[1] < k: # k 가 다리의 중량 제한이 넘으면 안됨
            continue
        else: # k 보다 크거나 같다면
            return dfs(i[0],y,k)

    return False

start = 1 # 시작 중량
end = 1000000000
def binary_search(start,end):
    ans = 1
    while start <= end:
        mid = (start + end) // 2

        if dfs(x,y,mid) == True: # mid 가 작거나 같다
            ans = mid
            start = mid+1
        else: # mid가 너무 크다
            end = mid-1

    return ans

print(binary_search(start,end))
```
