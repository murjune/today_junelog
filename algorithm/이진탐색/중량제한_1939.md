# dfs 함수
def dfs(visted,x,y,k): # x: 시작 섬, y: 끝 섬, k: 중량 기준
    if x == y:
        return True

    visted[x] = True # 방문 처리

    for i in graph[x]:
        if visted[i[0]] == True: # 이미 방문한 곳이면 안됨
            continue
        if i[1] < k: # k 가 다리의 중량 제한이 넘으면 안됨
            continue
        else: # k 보다 크거나 같다면
            return dfs(visted,i[0],y,k)

    return False
