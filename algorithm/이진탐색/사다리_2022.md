# 문제: 사다리_2022  
https://www.acmicpc.net/problem/2022


``` python


# 사다리 - 닮음 문제

# 절대/상대 오차는 10-3, 소수점 3번쨰 자리까지 출력

x, y, c = map(int,input().split())


# 풀이
# 1. x,y,c,ans1, ans2 의 관계식을 통해 ans를 구하자
# 2. x,y의 값이 너무 크니까 x: y를 자연수 값으로 나타내자.

# GCD - 최대 공약수

start = 1
end = min(x,y)

def GCD(start, end):
    ans = 1
    while start <= end:
        mid = (start + end) //2

        if x % mid == 0 and y % mid == 0:
            ans = mid
            start = mid +1

        else:
            end = mid -1

    return ans

print(GCD(start,end))




```
