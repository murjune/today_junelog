```python

# 마지막 과정인 [a1,a2]가 남았을 때, arr[mid] 는 무조건 a1가 된다.
# 그 이유 는 (s+e)// 2 즉, 소수점이 버림이 되기 때문이다.
# case 분류
# s: 0 , e : 0, arr[mid] = a1
# if a1 > target -> e =mid- 1(s:0,e:-1)-> loop종료

# if a1 < target -> s =mid+ 1(s: 1,e : 0)-> loop종료
# if # if a1 == target -> s =mid+ 1(s: 1,e : 0)-> loop종료

# 이분 탐색
# target이하의 원소를 뽑는 알고리즘
target = 1
arr= [1,3,4,7,9,12,30,40]

s = 0
e = len(arr)- 1

while s <= e:

    mid = (s+e)// 2
    if target >= arr[mid] :
        s = mid + 1
    elif target < arr[mid]:
        e = mid -1

print(arr[e])
# 이분 탐색
# target이상의 원소를 뽑는 알고리즘
target = 3
arr= [1,3,4,7,9,12,30,40]

s = 0
e = len(arr)- 1

while s <= e:

    mid = (s+e)// 2

    if target > arr[mid]:
        s = mid +1
    elif target <= arr[mid]:
        e = mid - 1

print(arr[s])
```
